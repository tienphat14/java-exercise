public static void main(String args[]) {
    boolean result = true;
    char multiple = '*';
    char c = 42;
    char c1 = 052;
    char c2 = 0x2a;
    char c3 = '\u002a'
    byte b = 100;
    short s = 10000;
    int i = 9876543210;
    int decVal = 26;
    int hexVal = 0x1a;
    int binVal = 0b11010;
    float f1  = 123.4f;
    float f2  = 1.234e2;
    double d1 = 123.4;
    double d2 = 1.234e2;
    long creditCardNumber = 1234_5678_9012_3456;
    long socialSecurityNumber = 999_99_9999L;
    float pi =  3.14_15F;
    float pi1 = 3_.1415F;
    float pi2 = 3._1415F;
    long socialSecurityNumber1 = 999_99_9999_L;
    int i = 1234567890;
    int x1 = 5_2;
    int x2 = 52_;
    int x3 = 5_______2;
    int x4 = 0_x52;
    int x5 = 0x_52;
    int x6 = 0x5_2;
    int x7 = 0x52_;
}
=> char c3 = '\u002a' -> Missed ';'
float pi1 = 3_.1415F; -> '_' next to '.'
float pi2 = 3._1415F; -> '_' next to '.'
int x2 = 52_; -> '_' placed at the end
int x4 = 0_x52; -> '_' placed at the end
int x5 = 0x_52; -> '_' placed at the beginning
int x7 = 0x52_; -> '_' placed at the end
