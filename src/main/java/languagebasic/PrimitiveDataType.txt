1. Find errors and fix them
public static void main(String args[]) {
    boolean boolean = false;
    System.out.println(boolean);
    boolean a = true;
    System.out.println(a);
    boolean b = 1;
    System.out.println(b);
    boolean c;
    System.out.println(c);
}
----------------------------------------------------------------
Error:
    boolean boolean = false; => Shouldn't name the variable same as data type

    boolean b = 1; => required type = boolean, but provided type = int

    boolean c; => don't have default value

Fix:
public static void main(String args[]) {
        boolean d = false;
        System.out.println(d);
        boolean a = true;
        System.out.println(a);
        boolean b = false;
        System.out.println(b);
        boolean c = true;
        System.out.println(c);
    }
----------------------------------------------------------------

public static void main(String args[]) {
    boolean result = true;
    char multiple = '*';
    char c = 42;
    char c1 = 052;
    char c2 = 0x2a;
    char c3 = '\u002a'
    byte b = 100;
    short s = 10000;
    int i = 9876543210;
    int decVal = 26;
    int hexVal = 0x1a;
    int binVal = 0b11010;
    float f1  = 123.4f;
    float f2  = 1.234e2;
    double d1 = 123.4;
    double d2 = 1.234e2;
    long creditCardNumber = 1234_5678_9012_3456;
    long socialSecurityNumber = 999_99_9999L;
    float pi =  3.14_15F;
    float pi1 = 3_.1415F;
    float pi2 = 3._1415F;
    long socialSecurityNumber1 = 999_99_9999_L;
    int i = 1234567890;
    int x1 = 5_2;
    int x2 = 52_;
    int x3 = 5_______2;
    int x4 = 0_x52;
    int x5 = 0x_52;
    int x6 = 0x5_2;
    int x7 = 0x52_;
}
----------------------------------------------------------------
Error:
        char c3 = '\u002a' => missing ;

        int i = 9876543210; => too big, maximum: 2147483647

        float f2  = 1.234e2; => missing "f"

        long creditCardNumber = 1234_5678_9012_3456L; => missing "L"

        float pi1 = 3_.1415F; => _ shouldn't be next to .
        float pi2 = 3._1415F;=> _ shouldn't be next to .
        long socialSecurityNumber1 = 999_99_9999_L; => _ shouldn't be next to L
        int i = 1234567890; => already defined

        int x2 = 52_; => _ shouldn't be at the end of number

        int x4 = 0_x52; => _ shouldn't be next to x
        int x5 = 0x_52; => _ shouldn't be next to x

        int x7 = 0x52_;  => _ shouldn't be at the end of number

4. What is the output?
public static void main(String args[]) {
    char c1 = 42;
    char c2 = '\u0005';
    System.out.println(c1 + c2); => because of "+" => char convert to int => 47
    System.out.println("**********");

    int myInt = 9;
    long myLong = myInt;
    double myDouble = myInt;
    System.out.println(myInt); => 9
    System.out.println(myLong); => 9
    System.out.println(myDouble); => 9.0
    System.out.println("**********");

    myDouble = 9.78;
    myInt = (int) myDouble;
    System.out.println(myDouble); => 9.78
    System.out.println(myInt); => 9
    System.out.println("**********");

    myLong = 9876543210L;
    myInt = (int) myLong;
    System.out.println(myLong); => 9876543210
    System.out.println(myInt); => 1286608618
    System.out.println("**********");

    float myFloat = 9.78F;
    myDouble = myFloat;
    System.out.println(myFloat); => 9.78
    System.out.println(myDouble); => 9.779999732971191
}
